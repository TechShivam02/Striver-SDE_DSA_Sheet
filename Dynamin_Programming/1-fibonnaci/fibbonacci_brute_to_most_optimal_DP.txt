
#include <iostream>
using namespace std;

int main()
{
    
    int n;
    cin>>n;
    
    int prev2=0;
    int prev1=1;
    
    int curr=prev2+prev1;
    
    for(int i=2;i<=n;i++){
        curr = prev2+prev1;
        
        prev2 = prev1;
        prev1 = curr;
        
    }
    
    cout<<"Fibo of "<<n<<" : "<<curr<<endl;
    

    return 0;
}
