    ListNode *detectCycle(ListNode *head) {
      
        if(head == NULL || head->next==head){
           return head;
       }
       
        ListNode *slow = head;
        ListNode *fast = head->next;
        
        bool ans = detectIfPresent(head , slow , fast);
        
        if(ans){  // it means cycle present
            
            slow = head;
            
            while(fast -> next != slow){
                slow = slow->next;
                fast=fast->next;
            }
            
            return slow;
        }
        
        return NULL;
    }
