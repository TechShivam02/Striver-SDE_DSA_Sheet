#include<iostream>
using namespace std;
        
        
struct node{
        
    int data;
    
    struct node* next;
    struct node* prev;
    
    node(int newData){
                
        this->data = newData;
        
        this->prev = NULL;
        this->next = NULL;
        
                
    }
        
};
        
        
        
struct node* head;
struct node* tail;

          
        
            
    void insertEnd(struct node* &tail , int newData){
        
               
               
                
              struct node* temp = new node(newData);
              tail->next = temp;
              temp ->prev = tail;
              
              tail = tail->next;

    }
            
            
        
    void display(struct node* head){
        
              while(head != NULL)
                {
                    cout<<head->data << " <-> ";
                    head = head->next;
                }
                
                cout<<endl<<endl;
        
        
    }

            {
                return;
            }

            node* temp = head;

            while (temp)
            {

                    node* ptr = head;

                    head=head->next;
                    temp=  head;

                    delete ptr;

                }

                else if(temp->next && temp->next->data == key){
                    
                    node* ptr = temp->next;
                    temp->next = temp->next->next;

                    delete ptr;

                }


                else{
                    temp = temp->next;
                }


            }



            display(head);
            

    }
