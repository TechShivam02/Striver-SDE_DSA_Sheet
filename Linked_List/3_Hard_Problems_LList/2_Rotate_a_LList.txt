
    int countNodes(Node* head){
    
        Node* temp = head;
        
        int count=0;
        
        
        while(temp){
            
            temp = temp->next;
            count++;
        }
        
        return count;
        
        
    }
    
    
    void insertEnd( Node* &tail , int data){
        
            Node* temp = new Node(data);
            tail->next = temp;
            tail= tail->next;
    }
    
    Node* rotate(Node* head, int k)
    {
        if(head == NULL){
            return head;
        }
        
        int count = countNodes(head);
        
        
        
        if(k==count){
            return head;
        }
        
        if(k>count){
            k = count % k;
        }
        
        Node* tail = head;
        
        while(tail->next != NULL){
            tail=tail->next;    
        }
        
        
        
        for(int i=0;i<k;i++){
        
            int data =  head->data;
            
            Node* temp = head;
            
            
            head = head->next;
            
            delete temp;
            
            insertEnd(tail , data);
            
        }
        
        return head;
        
        
    }
